60 def search(self, query: str, num: int = 3) -> List[Dict[str, Any]]: 

61 """検索を実行 

62 

63 戻り値の各要素: { 

64 title, url, snippet, source, published_at(ISO8601|None), score, reasons(list[str]) 

65 } 

66 """ 

67 config = self._get_search_config() 

68 provider = (config.get("provider") or "none").lower() 

69 

70 if provider == "none": 

71 results = self._search_none(query, num) 

72 elif provider == "stub": 

73 results = self._search_stub(query, num) 

74 elif provider == "cse": 

75 results = self._search_cse_with_fallback(query, num) 

76 elif provider == "newsapi": 

77 results = self._search_newsapi_with_fallback(query, num) 

78 elif provider == "hybrid": 

79 results = self._search_hybrid(query, num, config.get("limit", num)) 

80 else: 

81 results = self._search_unknown(query, num) 

82 

83 if not results: 

84 return [{ 

85 "title": "検索結果なし", 

86 "snippet": "ヒットしませんでした。別のキーワードで再検索してください。", 

87 "url": "", 

88 "source": "system", 

89 }] 

90 

91 if self.offline_mode: 

92 logger.warning("オフラインモード: Web検索に失敗したためスタブデータを使用します。") 

93 

94 return results

267 def _merge_dedupe(self, a: List[Dict[str, Any]], b: List[Dict[str, Any]], limit: int) -> List[Dict[str, Any]]: 

268 seen = set() 

269 merged: List[Dict[str, Any]] = [] 

270 for item in [*a, *b]: 

271 key = self._normalize_url(item.get("url")) 

272 if not key or key in seen: 

273 continue 

274 seen.add(key) 

275 merged.append(item) 

276 if len(merged) >= limit: 

277 break 

278 return merged

289 def _rank_results(self, items: List[Dict[str, Any]], query: str, num: int) -> List[Dict[str, Any]]: 

290 """高度化された検索結果のランキング""" 

291 if not items: 

292 return [] 

293 now = datetime.now(timezone.utc) 

294 cfg = self._get_search_config() 

295 keywords = [w for w in re.split(r"\s+", query) if len(w) >= 2] 

296 trusted_domains_map = {d: 1.0 for d in cfg.get("trusted_domains", [])} 

297 ranked: List[Dict[str, Any]] = [] 

298 

299 for it in items: 

300 score = 0.0 

301 reasons: List[str] = [] 

302 fresh = 0.0 # 新鮮度スコアを初期化 

303 

304 # 1. 新鮮度スコア（時効性の重要度を向上） 

305 ts = it.get("published_at") 

306 if ts: 

307 try: 

308 dt = datetime.fromisoformat(ts.replace("Z", "+00:00")) 

309 days = max(0, (now - dt).days) 

310 window = max(1, int(cfg.get("time_window_days", 60))) 

311 

312 # 非線形な新鮮度計算（より新しい記事により高い重み） 

313 if days <= 1: 

314 fresh = 1.0 # 24時間以内 

315 elif days <= 7: 

316 fresh = 0.9 # 1週間以内 

317 elif days <= 30: 

318 fresh = 0.7 # 1ヶ月以内 

319 elif days <= 90: 

320 fresh = 0.4 # 3ヶ月以内 

321 else: 

322 fresh = max(0.1, 1.0 - (days / float(window))) 

323 

324 score += fresh * 1.2 # 新鮮度の重みを増加 

325 reasons.append("freshness") 

326 except Exception: 

327 pass 

328 

329 # 2. ドメイン信頼性スコア（階層化された信頼度） 

330 host = urlparse(it.get("url", "")).netloc 

331 if host in trusted_domains_map: 

332 # 信頼ドメインの種類に応じた重み付け 

333 if any(high_trust in host for high_trust in ["www.nikkei.com", "www.bloomberg.co.jp", "www.reuters.com"]): 

334 score += 1.0 # 最高信頼度 

335 reasons.append("high_trusted_domain") 

336 elif any(medium_trust in host for medium_trust in ["techcrunch.com", "wired.com", "mit.edu"]): 

337 score += 0.8 # 中程度信頼度 

338 reasons.append("medium_trusted_domain") 

339 else: 

340 score += 0.6 # 基本信頼度 

341 reasons.append("trusted_domain") 

342 

343 # 3. クエリ関連性スコア（より洗練されたマッチング） 

344 text = ((it.get("title") or "") + " " + (it.get("snippet") or "")).lower() 

345 

346 # タイトルとスニペットの重み付け 

347 title_text = (it.get("title") or "").lower() 

348 snippet_text = (it.get("snippet") or "").lower() 

349 

350 title_matches = sum(1 for k in keywords if k.lower() in title_text) 

351 snippet_matches = sum(1 for k in keywords if k.lower() in snippet_text) 

352 

353 # タイトルマッチの方が重要 

354 relevance_score = (title_matches * 0.7 + snippet_matches * 0.3) / max(1, len(keywords)) 

355 score += min(1.0, relevance_score) * 0.8 

356 reasons.append("keyword_match") 

357 

358 # 4. コンテンツ品質スコア（新しい指標） 

359 content_quality = 0.0 

360 

361 # タイトルの長さ（適切な長さを評価） 

362 title_length = len(it.get("title", "")) 

363 if 20 <= title_length <= 100: 

364 content_quality += 0.2 

365 reasons.append("optimal_title_length") 

366 

367 # スニペットの詳細度 

368 snippet_length = len(it.get("snippet", "")) 

369 if snippet_length >= 100: 

370 content_quality += 0.2 

371 reasons.append("detailed_snippet") 

372 

373 # 特殊文字やHTMLタグの除去 

374 clean_text = re.sub(r'<[^>]+>', '', text) 

375 if clean_text == text: 

376 content_quality += 0.1 

377 reasons.append("clean_content") 

378 

379 score += content_quality 

380 reasons.append("content_quality") 

381 

382 # 5. ソース品質スコア 

383 source_quality = 0.0 

384 source = it.get("source", "") 

385 

386 if source == "newsapi": 

387 source_quality += 0.3 

388 reasons.append("news_api_source") 

389 elif source == "cse": 

390 source_quality += 0.2 

391 reasons.append("custom_search_source") 

392 

393 score += source_quality 

394 

395 # 6. 多様性ボーナス（同じドメインからの結果を抑制） 

396 domain_count = sum(1 for r in ranked if urlparse(r.get("url", "")).netloc == host) 

397 if domain_count == 0: 

398 score += 0.1 

399 reasons.append("diversity_bonus") 

400 

401 # スコアの正規化と最終調整 

402 final_score = min(5.0, score) # 最大スコアを5.0に制限 

403 

404 it["score"] = round(final_score, 3) 

405 it["reasons"] = reasons 

406 it["detailed_scoring"] = { 

407 "freshness": fresh, 

408 "reliability": score - fresh - content_quality - source_quality, 

409 "relevance": relevance_score * 0.8, 

410 "content_quality": content_quality, 

411 "source_quality": source_quality 

412 } 

413 

414 ranked.append(it) 

415 

416 # スコアによる並び替え 

417 ranked.sort(key=lambda x: x.get("score", 0), reverse=True) 

418 

419 # 結果の多様性を確保（上位結果のドメイン重複を制限） 

420 final_results = [] 

421 seen_domains = set() 

422 

423 # まず、ドメインが重複しない結果を優先的に追加 

424 for item in ranked: 

425 if len(final_results) >= num: 

426 break 

427 

428 host = urlparse(item.get("url", "")).netloc 

429 if host not in seen_domains: 

430 final_results.append(item) 

431 seen_domains.add(host) 

432 

433 # 要求された数に満たない場合は、残りの結果を追加 

434 if len(final_results) < num: 

435 for item in ranked: 

436 if len(final_results) >= num: 

437 break 

438 

439 # まだ追加されていない結果のみ 

440 if item not in final_results: 

441 final_results.append(item) 

442 

443 return final_results[:num]
