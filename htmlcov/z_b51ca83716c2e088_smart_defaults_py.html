102 return {}

210 def validate_form_data(self, form_data: Dict[str, Any], sales_style: SalesStyle) -> Dict[str, List[str]]: 

211 """ 

212 フォームデータをスマートデフォルトで検証 

213 

214 Args: 

215 form_data: フォームデータ 

216 sales_style: 営業スタイル 

217 

218 Returns: 

219 検証結果（提案と警告） 

220 """ 

221 defaults = self.get_smart_defaults(sales_style) 

222 suggestions = [] 

223 warnings = [] 

224 

225 # 目的の検証 

226 if not form_data.get("purpose"): 

227 suggestions.append(f"目的例: {defaults.get('purpose', '商談の目的を明確に設定しましょう')}") 

228 

229 # 制約の検証 

230 constraints = form_data.get("constraints", []) 

231 if not constraints: 

232 default_constraints = defaults.get("constraints", []) 

233 if default_constraints: 

234 suggestions.append(f"制約例: {', '.join(default_constraints[:2])}") 

235 

236 # 業界適合性の検証 

237 industry = form_data.get("industry", "") 

238 if industry and defaults.get("industry_focus"): 

239 if defaults["industry_focus"] not in industry and industry not in defaults["industry_focus"]: 

240 warnings.append(f"{industry}業界では{defaults['industry_focus']}の経験が有効です") 

241 

242 return { 

243 "suggestions": suggestions, 

244 "warnings": warnings 

245 }

248def apply_smart_defaults_to_form(form_data: Dict[str, Any], sales_style: SalesStyle, 

249 industry: str = "") -> Dict[str, Any]: 

250 """ 

251 フォームにスマートデフォルトを適用 

252 

253 Args: 

254 form_data: 現在のフォームデータ 

255 sales_style: 営業スタイル 

256 industry: 業界 

257 

258 Returns: 

259 更新されたフォームデータ 

260 """ 

261 manager = SmartDefaultsManager() 

262 defaults = manager.get_smart_defaults(sales_style, industry) 

263 

264 updated_data = form_data.copy() 

265 

266 # 空のフィールドにデフォルト値を適用 

267 for key, value in defaults.items(): 

268 if key in updated_data and not updated_data[key]: 

269 if isinstance(value, list): 

270 updated_data[key] = value[0] if value else "" 

271 else: 

272 updated_data[key] = value 

273 

274 return updated_data
