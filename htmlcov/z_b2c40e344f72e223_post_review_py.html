329def save_post_review(**kwargs) -> str: 

330 """商談後ふりかえり解析の結果をセッション形式で保存し、Session IDを返す""" 

331 try: 

332 provider = get_storage_provider() 

333 payload = { 

334 "type": "post_review", 

335 "input": { 

336 "sales_type": kwargs.get("sales_type"), 

337 "industry": kwargs.get("industry"), 

338 "product": kwargs.get("product"), 

339 "meeting_date": str(kwargs.get("meeting_date")), 

340 "meeting_duration": kwargs.get("meeting_duration"), 

341 "meeting_type": kwargs.get("meeting_type"), 

342 "meeting_content": kwargs.get("meeting_content"), 

343 "customer_reaction": kwargs.get("customer_reaction"), 

344 "challenges": kwargs.get("challenges"), 

345 "next_meeting": kwargs.get("next_meeting"), 

346 }, 

347 "output": { 

348 "analysis_result": kwargs.get("analysis_result"), 

349 }, 

350 } 

351 session_id = provider.save_session(payload) 

352 return session_id 

353 except Exception as e: 

354 st.error(f"保存に失敗しました: {str(e)}") 

355 raise
