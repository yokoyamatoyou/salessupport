25 """プロンプトテンプレートを読み込み""" 

26 try:

32 def generate_icebreakers(self, sales_type: SalesType, industry: str, company_hint: str = None, search_enabled: bool = True) -> List[str]: 

33 """アイスブレイクを生成""" 

34 try: 

35 # 業界ニュースを取得 

36 news_items = [] 

37 if search_enabled: 

38 news_items = self.search_provider.search(f"{industry} 最新ニュース", 2) 

39 # UI表示用に保持 

40 self.last_news_items = news_items or [] 

41 

42 # トーン辞書を取得 

43 tone = self._get_tone_for_type(sales_type) 

44 

45 # プロンプトを構築 

46 prompt = self._build_prompt(sales_type, industry, company_hint, news_items, tone) 

47 

48 # LLMで生成（プロバイダがない場合は例外→フォールバック） 

49 response = self.llm_provider.call_llm( 

50 prompt, "creative", self._get_icebreaker_schema() 

51 ) if self.llm_provider else None 

52 

53 # レスポンスからアイスブレイクを抽出 

54 if isinstance(response, dict) and "icebreakers" in response: 

55 return response["icebreakers"] 

56 else: 

57 # フォールバック: レスポンスが期待される形式でない場合 

58 return self._generate_fallback_icebreakers(sales_type, industry, tone) 

59 

60 except Exception as e: 

61 # エラーが発生した場合はフォールバックを使用 

62 tone = self._get_tone_for_type(sales_type) 

63 # 検索結果はリセット 

64 self.last_news_items = [] 

65 return self._generate_fallback_icebreakers(sales_type, industry, tone)

82 def _build_prompt(self, sales_type: SalesType, industry: str, company_hint: str, news_items: List[Dict], tone: str) -> str: 

83 """プロンプトを構築""" 

84 # システムメッセージ 

85 system_msg = self.prompt_template["system"] 

86 

87 # 業界ニュースの詳細を文字列として構築 

88 news_details = "" 

89 if news_items: 

90 news_details = "\n".join( 

91 [ 

92 f"- {escape_braces(sanitize_for_prompt(item['title']))}: {escape_braces(sanitize_for_prompt(item['snippet']))}" 

93 for item in news_items 

94 ] 

95 ) 

96 news_details = sanitize_for_prompt(news_details) 

97 

98 # ユーザーメッセージ 

99 user_template = Template(self.prompt_template["user_template"]) 

100 user_msg = user_template.safe_substitute( 

101 sales_type=escape_braces(sanitize_for_prompt(sales_type.value)), 

102 tone=escape_braces(sanitize_for_prompt(tone)), 

103 industry=escape_braces(sanitize_for_prompt(industry)), 

104 company_hint=escape_braces(sanitize_for_prompt(company_hint or 'なし')), 

105 news_items=escape_braces(sanitize_for_prompt(news_details)), 

106 ) 

107 

108 # 出力制約 

109 output_constraints = "\n".join(self.prompt_template["output_constraints"]) 

110 

111 # 完全なプロンプトを構築 

112 full_prompt = f""" 

113{system_msg} 

114 

115{user_msg} 

116 

117出力制約: 

118{output_constraints} 

119 

120上記の情報を基に、{tone}のトーンで1行のアイスブレイクを3つ生成してください。 

121各アイスブレイクは自然で親しみやすく、商談の導入として適切な内容にしてください。 

122""" 

123 

124 return full_prompt.replace("{{", "{").replace("}}", "}")
