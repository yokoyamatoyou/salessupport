55class FileBasedSchemaStorage(SchemaStorageInterface): 

56 """ファイルベースのスキーマストレージ""" 

57 

58 def __init__(self, base_path: Path): 

59 self.base_path = base_path 

60 self.schemas: Dict[str, Dict[str, SchemaDefinition]] = {} 

61 self._load_all_schemas() 

62 

63 def _load_all_schemas(self): 

64 """全てのスキーマを読み込み""" 

65 if not self.base_path.exists(): 

66 self.base_path.mkdir(parents=True, exist_ok=True) 

67 logger.info(f"Created schema directory: {self.base_path}") 

68 return 

69 

70 for json_file in self.base_path.glob("*.json"): 

71 try: 

72 self._load_schema_from_file(json_file) 

73 except Exception as e: 

74 logger.error(f"Failed to load schema {json_file}: {e}") 

75 

76 def _load_schema_from_file(self, file_path: Path): 

77 """ファイルからスキーマを読み込み""" 

78 with open(file_path, 'r', encoding='utf-8') as f: 

79 data = json.load(f) 

80 

81 schema_def = SchemaDefinition( 

82 name=data['name'], 

83 version=data['version'], 

84 schema=data['schema'], 

85 description=data.get('description', ''), 

86 tags=data.get('tags', []), 

87 metadata=data.get('metadata', {}) 

88 ) 

89 

90 if schema_def.name not in self.schemas: 

91 self.schemas[schema_def.name] = {} 

92 

93 self.schemas[schema_def.name][schema_def.version] = schema_def 

94 logger.info(f"Loaded schema: {schema_def.name} v{schema_def.version}") 

95 

96 def load_schema(self, name: str, version: Optional[str] = None) -> Optional[SchemaDefinition]: 

97 if name not in self.schemas: 

98 return None 

99 

100 versions = self.schemas[name] 

101 

102 # バージョン指定がない場合は最新版を返す 

103 if version is None: 

104 latest_version = max(versions.keys()) 

105 return versions[latest_version] 

106 

107 return versions.get(version) 

108 

109 def save_schema(self, schema_def: SchemaDefinition) -> None: 

110 file_name = f"{schema_def.name}_v{schema_def.version}.json" 

111 file_path = self.base_path / file_name 

112 

113 data = { 

114 'name': schema_def.name, 

115 'version': schema_def.version, 

116 'schema': schema_def.schema, 

117 'description': schema_def.description, 

118 'tags': schema_def.tags, 

119 'metadata': schema_def.metadata 

120 } 

121 

122 with open(file_path, 'w', encoding='utf-8') as f: 

123 json.dump(data, f, ensure_ascii=False, indent=2) 

124 

125 if schema_def.name not in self.schemas: 

126 self.schemas[schema_def.name] = {} 

127 

128 self.schemas[schema_def.name][schema_def.version] = schema_def 

129 logger.info(f"Saved schema: {schema_def.name} v{schema_def.version}") 

130 

131 def list_schemas(self) -> List[str]: 

132 return list(self.schemas.keys()) 

133 

134 def exists(self, name: str, version: Optional[str] = None) -> bool: 

135 if name not in self.schemas: 

136 return False 

137 

138 if version is None: 

139 return len(self.schemas[name]) > 0 

140 

141 return version in self.schemas[name]

327class UnifiedSchemaManager: 

328 """統一されたスキーママネージャー""" 

329 

330 def __init__(self, storage: Optional[SchemaStorageInterface] = None): 

331 self.storage = storage or FileBasedSchemaStorage(Path("schemas")) 

332 self.builder = SchemaBuilder() 

333 self.logger = Logger("UnifiedSchemaManager") 

334 self._initialize_default_schemas() 

335 

336 def _initialize_default_schemas(self): 

337 """デフォルトスキーマを初期化""" 

338 default_schemas = [ 

339 ("pre_advice", "1.0.0", self.builder.create_pre_advice_schema(), "事前アドバイス出力スキーマ"), 

340 ("post_review", "1.0.0", self.builder.create_post_review_schema(), "商談後ふりかえりスキーマ"), 

341 ("icebreaker", "1.0.0", self.builder.create_icebreaker_schema(), "アイスブレイカースキーマ") 

342 ] 

343 

344 for name, version, schema, description in default_schemas: 

345 if not self.storage.exists(name, version): 

346 schema_def = SchemaDefinition( 

347 name=name, 

348 version=version, 

349 schema=schema, 

350 description=description, 

351 tags=["default", "system"] 

352 ) 

353 try: 

354 self.storage.save_schema(schema_def) 

355 self.logger.info(f"Initialized default schema: {name}") 

356 except Exception as e: 

357 self.logger.error(f"Failed to initialize schema {name}: {e}") 

358 

359 def get_schema(self, name: str, version: Optional[str] = None) -> Optional[Dict[str, Any]]: 

360 """スキーマを取得""" 

361 schema_def = self.storage.load_schema(name, version) 

362 return schema_def.schema if schema_def else None 

363 

364 def create_schema(self, name: str, schema: Dict[str, Any], version: str = "1.0.0", 

365 description: str = "", tags: List[str] = None) -> None: 

366 """スキーマを作成""" 

367 if self.storage.exists(name, version): 

368 raise ConfigurationError(f"Schema already exists: {name} v{version}") 

369 

370 schema_def = SchemaDefinition( 

371 name=name, 

372 version=version, 

373 schema=schema, 

374 description=description, 

375 tags=tags or [] 

376 ) 

377 

378 self.storage.save_schema(schema_def) 

379 self.logger.info(f"Created schema: {name} v{version}") 

380 

381 def update_schema(self, name: str, schema: Dict[str, Any], version: str, 

382 description: Optional[str] = None) -> None: 

383 """スキーマを更新""" 

384 if not self.storage.exists(name): 

385 raise ConfigurationError(f"Schema not found: {name}") 

386 

387 schema_def = SchemaDefinition( 

388 name=name, 

389 version=version, 

390 schema=schema, 

391 description=description or "", 

392 tags=["updated"] 

393 ) 

394 

395 self.storage.save_schema(schema_def) 

396 self.logger.info(f"Updated schema: {name} v{version}") 

397 

398 def list_available_schemas(self) -> List[Dict[str, Any]]: 

399 """利用可能なスキーマ一覧を取得""" 

400 schemas = [] 

401 for name in self.storage.list_schemas(): 

402 schema_def = self.storage.load_schema(name) 

403 if schema_def: 

404 schemas.append({ 

405 'name': schema_def.name, 

406 'version': schema_def.version, 

407 'description': schema_def.description, 

408 'tags': schema_def.tags 

409 }) 

410 return schemas 

411 

412 def validate_schema_definition(self, schema: Dict[str, Any]) -> List[str]: 

413 """スキーマ定義の妥当性を検証""" 

414 errors = [] 

415 

416 if not isinstance(schema, dict): 

417 errors.append("Schema must be a dictionary") 

418 return errors 

419 

420 if 'type' not in schema: 

421 errors.append("Schema must have a 'type' field") 

422 

423 # 基本的なJSON Schema検証 

424 if schema.get('type') not in ['object', 'array', 'string', 'number', 'boolean']: 

425 errors.append("Invalid schema type") 

426 

427 return errors
